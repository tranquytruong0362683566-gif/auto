// HÀNH ĐỘNG CLICK ĐĂNG REELS (GỬI SANG BACKEND) document.getElementById('btnSubmitReels').addEventListener('click', async () => { // 1. Kiểm tra Cookie từ Extension if (!userCookie) { alert("❌ Chưa có Cookie Facebook! Web chưa kết nối được với Extension hoặc bạn chưa đăng nhập Facebook."); return; } // 2. Lấy dữ liệu từ form const selectedPages = Array.from(document.querySelectorAll('.row-check:checked')).map(cb => cb.value); const content = document.getElementById('postContent').value; const fileInput = document.getElementById('videoFile'); if (selectedPages.length === 0) { alert("⚠️ Vui lòng chọn ít nhất 1 Page ở bảng danh sách!"); return; } if (fileInput.files.length === 0) { alert("⚠️ Vui lòng chọn file Video để đăng!"); return; } // 3. Đóng gói dữ liệu thành FormData (Bắt buộc vì có chứa file vật lý) const formData = new FormData(); formData.append('cookie', userCookie); // Gửi kèm cookie lấy từ extension formData.append('pageIds', JSON.stringify(selectedPages)); // Chuyển mảng page sang chuỗi JSON formData.append('content', content); formData.append('video', fileInput.files[0]); // Đính kèm file video người dùng chọn // Thay đổi trạng thái nút bấm const btnSubmit = document.getElementById('btnSubmitReels'); btnSubmit.disabled = true; btnSubmit.textContent = '⏳ Backend đang xử lý đăng bài...'; btnSubmit.style.background = '#fbbc05'; try { // GỬI REQUEST SANG BACKEND SERVER ĐANG CHẠY Ở LOCALHOST const response = await fetch('http://localhost:3000/api/upload-reels', { method: 'POST', body: formData // Truyền toàn bộ dữ liệu qua body }); const result = await response.json(); if (result.success) { alert(`🎉 Thành công: ${result.message}`); modal.style.display = 'none'; // Đóng bảng cấu hình đăng bài } else { alert(`❌ Thất bại: ${result.error}`); } } catch (err) { console.error("Lỗi kết nối Backend:", err); alert("❌ Không thể kết nối tới Backend Server! Hãy chắc chắn rằng bạn đã chạy lệnh 'node server.js' ở máy tính."); } finally { // Khôi phục trạng thái nút bấm ban đầu btnSubmit.disabled = false; btnSubmit.style.background = ''; btnSubmit.innerHTML = `Bắt đầu đăng Reels (${selectedPages.length} trang)`; } });
